69class Semaphore {
70 private:
71 void* m_hSema;
72
73 Semaphore(const Semaphore& other) MOODYCAMEL_DELETE_FUNCTION;
74 Semaphore& operator=(const Semaphore& other) MOODYCAMEL_DELETE_FUNCTION;
75
76 public:
77 Semaphore(int initialCount = 0) {
78 assert(initialCount >= 0);
79 const long maxLong = 0x7fffffff;
80 m_hSema = CreateSemaphoreW(nullptr, initialCount, maxLong, nullptr);
81 assert(m_hSema);
82 }
83
84 ~Semaphore() { CloseHandle(m_hSema); }
85
86 bool wait() {
87 const unsigned long infinite = 0xffffffff;
88 return WaitForSingleObject(m_hSema, infinite) == 0;
89 }

91 bool try_wait() { return WaitForSingleObject(m_hSema, 0) == 0; }
92
93 bool timed_wait(std::uint64_t usecs) {
94 return WaitForSingleObject(m_hSema, (unsigned long)(usecs / 1000)) == 0;
95 }

97 void signal(int count = 1) {
98 while (!ReleaseSemaphore(m_hSema, count, nullptr))
99 ;

157class Semaphore {
158 private:
159 sem_t m_sema;
160
161 Semaphore(const Semaphore& other) MOODYCAMEL_DELETE_FUNCTION;
162 Semaphore& operator=(const Semaphore& other) MOODYCAMEL_DELETE_FUNCTION;
163
164 public:
165 Semaphore(int initialCount = 0) {
166 assert(initialCount >= 0);
167 int rc = sem_init(&m_sema, 0, static_cast<unsigned int>(initialCount));
168 assert(rc == 0);
169 (void)rc;
170 }
171
172 ~Semaphore() { sem_destroy(&m_sema); }
173
174 bool wait() {
175 // http://stackoverflow.com/questions/2013181/gdb-causes-sem-wait-to-fail-with-eintr-error
176 int rc;
177 do {
178 rc = sem_wait(&m_sema);
179 } while (rc == -1 && errno == EINTR);
180 return rc == 0;
181 }
182
183 bool try_wait() {
184 int rc;
185 do {
186 rc = sem_trywait(&m_sema);
187 } while (rc == -1 && errno == EINTR);
188 return rc == 0;
189 }
190
191 bool timed_wait(std::uint64_t usecs) {
192 struct timespec ts;
193 const int usecs_in_1_sec = 1000000;
194 const int nsecs_in_1_sec = 1000000000;
195#ifdef MOODYCAMEL_LIGHTWEIGHTSEMAPHORE_MONOTONIC
196 clock_gettime(CLOCK_MONOTONIC, &ts);
197#else
198 clock_gettime(CLOCK_REALTIME, &ts);
199#endif
200 ts.tv_sec += (time_t)(usecs / usecs_in_1_sec);
201 ts.tv_nsec += (long)(usecs % usecs_in_1_sec) * 1000;
202 // sem_timedwait bombs if you have more than 1e9 in tv_nsec
203 // so we have to clean things up before passing it in
204 if (ts.tv_nsec >= nsecs_in_1_sec) {
205 ts.tv_nsec -= nsecs_in_1_sec;
206 ++ts.tv_sec;
207 }
208
209 int rc;
210 do {
211#ifdef MOODYCAMEL_LIGHTWEIGHTSEMAPHORE_MONOTONIC
212 rc = sem_clockwait(&m_sema, CLOCK_MONOTONIC, &ts);
213#else
214 rc = sem_timedwait(&m_sema, &ts);
215#endif
216 } while (rc == -1 && errno == EINTR);
217 return rc == 0;
218 }
219
220 void signal() {
221 while (sem_post(&m_sema) == -1)
222 ;
223 }
224
225 void signal(int count) {
226 while (count-- > 0) {
227 while (sem_post(&m_sema) == -1)
228 ;
229 }
230 }
231};

241class LightweightSemaphore {
242 public:
243 typedef std::make_signed<std::size_t>::type ssize_t;
244
245 private:
246 std::atomic<ssize_t> m_count;
247 details::Semaphore m_sema;
248 int m_maxSpins;
249
250 bool waitWithPartialSpinning(std::int64_t timeout_usecs = -1) {
251 ssize_t oldCount;
252 int spin = m_maxSpins;
253 while (--spin >= 0) {
254 oldCount = m_count.load(std::memory_order_relaxed);
255 if ((oldCount > 0) &&
256 m_count.compare_exchange_strong(oldCount, oldCount - 1,
257 std::memory_order_acquire,
258 std::memory_order_relaxed))
259 return true;
260 std::atomic_signal_fence(
261 std::memory_order_acquire); // Prevent the compiler from collapsing
262 // the loop.
263 }
264 oldCount = m_count.fetch_sub(1, std::memory_order_acquire);
265 if (oldCount > 0) return true;
266 if (timeout_usecs < 0) {
267 if (m_sema.wait()) return true;
268 }
269 if (timeout_usecs > 0 && m_sema.timed_wait((std::uint64_t)timeout_usecs))
270 return true;
271 // At this point, we've timed out waiting for the semaphore, but the
272 // count is still decremented indicating we may still be waiting on
273 // it. So we have to re-adjust the count, but only if the semaphore
274 // wasn't signaled enough times for us too since then. If it was, we
275 // need to release the semaphore too.
276 while (true) {
277 oldCount = m_count.load(std::memory_order_acquire);
278 if (oldCount >= 0 && m_sema.try_wait()) return true;
279 if (oldCount < 0 && m_count.compare_exchange_strong(
280 oldCount, oldCount + 1, std::memory_order_relaxed,
281 std::memory_order_relaxed))
282 return false;
283 }
284 }
285
286 ssize_t waitManyWithPartialSpinning(ssize_t max,
287 std::int64_t timeout_usecs = -1) {
288 assert(max > 0);
289 ssize_t oldCount;
290 int spin = m_maxSpins;
291 while (--spin >= 0) {
292 oldCount = m_count.load(std::memory_order_relaxed);
293 if (oldCount > 0) {
294 ssize_t newCount = oldCount > max ? oldCount - max : 0;
295 if (m_count.compare_exchange_strong(oldCount, newCount,
296 std::memory_order_acquire,
297 std::memory_order_relaxed))
298 return oldCount - newCount;
299 }
300 std::atomic_signal_fence(std::memory_order_acquire);
301 }
302 oldCount = m_count.fetch_sub(1, std::memory_order_acquire);
303 if (oldCount <= 0) {
304 if ((timeout_usecs == 0) || (timeout_usecs < 0 && !m_sema.wait()) ||
305 (timeout_usecs > 0 &&
306 !m_sema.timed_wait((std::uint64_t)timeout_usecs))) {
307 while (true) {
308 oldCount = m_count.load(std::memory_order_acquire);
309 if (oldCount >= 0 && m_sema.try_wait()) break;
310 if (oldCount < 0 &&
311 m_count.compare_exchange_strong(oldCount, oldCount + 1,
312 std::memory_order_relaxed,
313 std::memory_order_relaxed))
314 return 0;
315 }
316 }
317 }
318 if (max > 1) return 1 + tryWaitMany(max - 1);
319 return 1;
320 }
321
322 public:
323 LightweightSemaphore(ssize_t initialCount = 0, int maxSpins = 10000)
324 : m_count(initialCount), m_maxSpins(maxSpins) {
325 assert(initialCount >= 0);
326 assert(maxSpins >= 0);
327 }
328
329 bool tryWait() {
330 ssize_t oldCount = m_count.load(std::memory_order_relaxed);
331 while (oldCount > 0) {
332 if (m_count.compare_exchange_weak(oldCount, oldCount - 1,
333 std::memory_order_acquire,
334 std::memory_order_relaxed))
335 return true;
336 }
337 return false;
338 }
339
340 bool wait() { return tryWait() || waitWithPartialSpinning(); }
341
342 bool wait(std::int64_t timeout_usecs) {
343 return tryWait() || waitWithPartialSpinning(timeout_usecs);
344 }
345
346 // Acquires between 0 and (greedily) max, inclusive
347 ssize_t tryWaitMany(ssize_t max) {
348 assert(max >= 0);
349 ssize_t oldCount = m_count.load(std::memory_order_relaxed);
350 while (oldCount > 0) {
351 ssize_t newCount = oldCount > max ? oldCount - max : 0;
352 if (m_count.compare_exchange_weak(oldCount, newCount,
353 std::memory_order_acquire,
354 std::memory_order_relaxed))
355 return oldCount - newCount;
356 }
357 return 0;
358 }
359
360 // Acquires at least one, and (greedily) at most max
361 ssize_t waitMany(ssize_t max, std::int64_t timeout_usecs) {
362 assert(max >= 0);
363 ssize_t result = tryWaitMany(max);
364 if (result == 0 && max > 0)
365 result = waitManyWithPartialSpinning(max, timeout_usecs);
366 return result;
367 }
368
369 ssize_t waitMany(ssize_t max) {
370 ssize_t result = waitMany(max, -1);
371 assert(result > 0);
372 return result;
373 }
374
375 void signal(ssize_t count = 1) {
376 assert(count >= 0);
377 ssize_t oldCount = m_count.fetch_add(count, std::memory_order_release);
378 ssize_t toRelease = -oldCount < count ? -oldCount : count;
379 if (toRelease > 0) {
380 m_sema.signal((int)toRelease);
381 }
382 }
383
384 std::size_t availableApprox() const {
385 ssize_t count = m_count.load(std::memory_order_relaxed);
386 return count > 0 ? static_cast<std::size_t>(count) : 0;
387 }
388};
